\[g'(t)=6\e^{-t}\left (1-t\right ).\]

\begin{enumerate}
\item  Étudier le signe de $g'(t)$ sur $[0\;;\;+\infty[$.
\item En déduire les variations de $g$ sur $[0\;;\;+\infty[$.
\end{enumerate}
