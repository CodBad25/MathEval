	\section*{Exercice 2 (5 points)}
		
\begin{enumerate}
	\item $f(0) = 1, \quad f(-2) = 5, \quad f'(0) = f'(-2) = 0.$
	
	\item \begin{enumerate}
	\item	$f'(x) = 3ax^2 + 2bx + c.$
	\item 
	
	\[
	\begin{cases}
		f(0) = 1 \\
		f'(0) = 0 \\
	\end{cases}
	\Rightarrow
	\begin{cases}
		d = 1 \\
		c = 0 \\
	\end{cases}
	\Rightarrow f(x) = ax^3 + bx^2 + 1.
	\]
	
	\item De même,
	
	\[
	\begin{cases}
		f(-2) = 5 \\
		f'(-2) = 0 \\
	\end{cases}
	\Rightarrow
	\begin{cases}
		-8a + 4b + 1 = 5 \\
		12a - 4b = 0 \\
	\end{cases}.
	\]
	
		\item Le système précédent s’écrit
	
	\[
	\begin{cases}
		-8a + 4b = 4 \\
		12a - 4b = 0 \\
	\end{cases}
	\Rightarrow
	\begin{cases}
		-2a + b = 1 \\
		3a - b = 0 \\
	\end{cases}
	\Rightarrow a = 1, \quad b = 3.
	\]
	
	Donc finalement : \(f(x) = x^3 + 3x^2 + 1\).
\end{enumerate}
\end{enumerate}
