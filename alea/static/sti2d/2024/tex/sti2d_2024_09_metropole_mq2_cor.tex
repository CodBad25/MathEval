\begin{list}{\textbullet}{}
\item $u(x) = x^2 \quad \text{et} \quad u'(x) = 2x$,
\item $v(x) = \e^{-2x} \quad \text{et} \quad v'(x) = -2e^{-2x}$.
\end{list}
\begin{align*}
g'(x) &= u'(x) \times v(x) + u(x) \times v'(x) \\
&= 2x \times \e^{-2x} + x^2 \times \left(-2\e^{-2x}\right) \\
&= 2x \e^{-2x} - 2x^2 \e^{-2x} \\
&= 2x \e^{-2x} (1 - x).
\end{align*}
