\begin{enumerate}[start=5]
\item On veut déterminer la fonction solution de l'équation différentielle $(E)$ s'annulant en $0$.

\begin{list}{\textbullet}{}
\item D'après le cours, l'équation différentielle $y'=-9y$ a pour solutions les fonctions $f$ définies par $f(t)=k \e^{-9t}$ où $k\in\R$.
\item L'équation $y'=-9y+8,6$ a pour solution constante la fonction $t \longmapsto\dfrac{8,6}{9}$ soit environ $0,96$.
\item L'équation $y'=-9y+8,6$ a donc pour solutions les fonctions $f$ définies par\\ $f(t)=k\e^{-9t}+0,96$, où $k\in\R$.
\item La solution qui s'annule en 0 vérifie $k\e^{0}+0,96=0$, ce qui entraine que $k=-9,6$.
\end{list}

La fonction solution cherchée est donc la fonction $f$ définie par $f(t)=0,96 - 0,96\e^{-9t}$.

\item $\ds\lim_{t\to +\infty} -9t = -\infty$ et $\ds\lim_{T\to -\infty} \e^{T}=0$ donc $\ds\lim_{t\to +\infty} \e^{-9t}=0$.

On en déduit que $\ds\lim_{t\to +\infty} 1-\e^{-9t}=1$ et donc que $\ds\lim_{t\to +\infty} 0,96\left ( 1-\e^{-9t}\right ) =0,96$.
\end{enumerate}
