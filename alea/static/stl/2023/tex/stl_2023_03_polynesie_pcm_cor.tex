\begin{enumerate}[start=5]
\item Les solutions de l'équation différentielle \og sans second membre \fg{} :
\[y'=-27,7y,\]

sont les fonctions $t \longmapsto k\times \e^{-27,7t}$ où $k\in\R$.

Une solution particulière de $(E)$ est la fonction constante $t \longmapsto \dfrac{9,81}{27,7}$.

Donc l'équation différentielle $(E)$ a pour solutions les fonctions $v$ définies par :
\[v(t) = k \times \e^{-27,7t} + \dfrac{9,81}{27,7} \text{ où } k\in\R.\]

\item $v(0) = 0 \iff k\times\e^{0}+\dfrac{9,81}{27,7} = 0 \iff k = -\dfrac{9,81}{27,7}$

Donc l'unique solution $v$ de l'équation différentielle $(E)$ qui vérifie $v(0) = 0$ est définie par l'expression : 
\[v(t)= -\dfrac{9,81}{27,7} \times \e^{-27,7t} + \dfrac{9,81}{27,7},\]

soit :
\[v(t) =\dfrac{9,81}{27,7}\times \left(1 -\e^{-27,7t}\right).\]

\item $\ds\lim_{t\to +\infty} -27,7t = -\infty$ et $\ds\lim_{T\to -\infty} \e^{T}=0$ ; donc $\ds\lim_{t\to +\infty} \e^{-27,7t} =0$.

On en déduit que :
$\ds\lim_{t\to +\infty} \left (1-\e^{-27,7t}\right )=1$ et donc que
$\ds\lim_{t\to +\infty} v(t) = \dfrac{9,81}{27,7}$.

\end{enumerate}
